39 /* NOTE!!
40  * The configFRTOS_MEMORY_SCHEME macro describes the heap scheme using a value
41  * 1 - 5 which corresponds to the following schemes:
42  *
43  * heap_1 - the very simplest, does not permit memory to be freed
44  * heap_2 - permits memory to be freed, but not does coalescence adjacent free
45  * blocks.
46  * heap_3 - simply wraps the standard malloc() and free() for thread safety
47  * heap_4 - coalesces adjacent free blocks to avoid fragmentation. Includes
48  * absolute address placement option
49  * heap_5 - as per heap_4, with the ability to span the heap across
50  * multiple nonOadjacent memory areas
51  * heap_6 - reentrant newlib implementation
52  */
53 #ifndef configUSE_HEAP_SCHEME
54  #define configUSE_HEAP_SCHEME 3 /* thread safe malloc */
55 #endif
56
